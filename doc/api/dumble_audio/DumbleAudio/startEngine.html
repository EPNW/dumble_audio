static Future<bool> startEngine(
    {AudioFormat? recordingFormat, AudioFormat? playingFormat}) async {
  if (!_hasPermission) {
    print('No Permission');
    return false;
  }
  final AudioFormat rFormat = recordingFormat ?? AudioFormat.standard();
  final AudioFormat pFormat = playingFormat ?? AudioFormat.standard();
  if (Platform.isIOS && pFormat.encoding == AudioEncoding.PCM_16BIT) {
    print('PCM_16BIT not supported on iOS for playing up to now');
    return false;
  }
  _eventChannel
      .receiveBroadcastStream()
      .map<Uint8List>((event) => event as Uint8List)
        ..listen((event) {
          RecordingCallback? recordingCallback = _recordingCallback;
          if (recordingCallback != null) {
            recordingCallback(event);
          }
        });
  final Map<String, dynamic> arguments = {};
  arguments.addAll(rFormat.toMap(true));
  arguments.addAll(pFormat.toMap(false));
  await _methodChannel.invokeMethod('startEngine', arguments);
  return true;
}
